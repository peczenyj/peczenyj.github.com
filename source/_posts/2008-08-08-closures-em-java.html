Olha o que vai ter no java 1.7

public class A {
        public static void main(String args[]) {
                exec( { int a, int b => a + b });
                exec( { int a, int b => a * b });
                exec( { int a, int b => 
                        System.out.println("closure " + a + ":" + b); 
                        (a * a) + (b * b) 
                });
        }
        public static void exec({ int, int => int } closure){
                System.out.println(closure.invoke(3,2));
        }
}
